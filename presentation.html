from langchain_openai import ChatOpenAI
from langgraph.prebuilt import create_react_agent
from config import ACTIVE_API_KEY, ACTIVE_BASE_URL, ACTIVE_MODEL
from tools import get_weather, search_train_ticket, draft_email

def create_travel_agent():
    # 1. 初始化 LLM（连接 MiMo API）
    llm = ChatOpenAI(
        model=ACTIVE_MODEL,          # mimo-v2.5-pro
        api_key=ACTIVE_API_KEY,      # API 密钥
        base_url=ACTIVE_BASE_URL,    # OpenAI 兼容接口
        temperature=0.7,             # 创造性参数
        streaming=True,              # 启用流式输出
    )

    # 2. 注册工具
    tools = [get_weather, search_train_ticket, draft_email]

    # 3. 系统提示词
    system_prompt = """你是一个专业的旅行助手Agent..."""

    # 4. 创建 ReAct Agent
    agent = create_react_agent(
        model=llm,
        tools=tools,
        prompt=system_prompt,
    )
    return agent

关键技术点

  • ChatOpenAI：LangChain 的 LLM 封装，兼容所有 OpenAI 格式 API
  • create_react_agent：LangGraph 提供的 ReAct 模式 Agent，自动进行"思考→行动→观察"循环
  • streaming=True：启用流式输出，实现逐字显示效果

import queue, threading

def run_single_agent(agent_type, config, user_prompt, q):
    """单个Agent运行线程 - 推送到队列"""
    try:
        q.put(("agent_start", {"agent_key": agent_type, ...}))

        if config["use_tools"]:
            # 使用工具的 Agent（ReAct 模式）
            agent = create_react_agent(model=llm, tools=tools, ...)
            for event in agent.stream(..., stream_mode="messages"):
                msg_obj = event[0]
                if hasattr(msg_obj, "tool_calls") and msg_obj.tool_calls:
                    q.put(("tool_call", {...}))       # 工具调用
                elif type(msg_obj).__name__ == "ToolMessage":
                    q.put(("tool_result", {...}))     # 工具结果
                elif msg_obj.content:
                    q.put(("chunk", {"content": msg_obj.content}))  # 流式文本
        else:
            # 直接调用 LLM（无工具）
            for chunk in llm.stream(messages):
                q.put(("chunk", {"content": chunk.content}))

        q.put(("agent_done", {...}))
    except Exception as e:
        q.put(("agent_error", {...}))

# 主函数：启动所有 Agent 线程

def generate():
    q = queue.Queue()
    threads = []

    # 启动所有 Agent 线程
    for agent_type in agents:
        t = threading.Thread(
            target=run_single_agent,
            args=(agent_type, config, prompt, q)
        )
        t.start()
        threads.append(t)

数据协议
SSE 事件类型定义
前后端通过统一的 SSE 事件协议进行通信
事件类型 字段 说明 触发时机
agent_start agent_key, agent_name Agent 开始工作 线程启动时
tool_call agent_key, tool_name, tool_args Agent 决定调用工具 ReAct 循环中
tool_result agent_key, tool_name, content 工具返回结果 工具执行完毕
chunk agent_key, content AI 流式文本片段 LLM 逐字输出
agent_done agent_key, agent_name Agent 完成工作 线程结束时
agent_error agent_key, error Agent 执行出错 异常捕获时
final content 所有 Agent 完成 全部线程结束

关键技术决策

  • threading vs asyncio：选择 threading 是因为 LangChain 的 LLM 调用是同步阻塞的，threading 更简单直接
  • Queue vs 直接 yield：使用 Queue 解耦生产者和消费者，避免线程竞争
  • SSE vs WebSocket：SSE 是单向推送，实现更简单，适合本场景的"服务器→客户端"数据流
